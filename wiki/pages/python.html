Format integer with leading zeros

print("{:02d}".format(5))

Format float with fixed decimal places

print("{:.2f}".format(2.7913))

Save object using pickle

import pickle
f = open("/tmp/a.pkl", "wb")
pickle.dump(obj, f, pickle.HIGHEST_PROTOCOL)
f.close()

Load object using pickle

import pickle
f = open("/tmp/a.pkl", "rb")
obj = pickle.load(f)
f.close()
